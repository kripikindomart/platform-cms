## Pendahuluan

Dokumen ini menjelaskan spesifikasi lengkap untuk **CLI Builder Tool** - sebuah command-line tool untuk mempercepat development dengan code generation otomatis. CLI tool ini akan dibangun pada **Week 12-14** setelah pattern development sudah established.

**Tujuan CLI Builder**:
- 🚀 Mempercepat development (reduce boilerplate)
- 📝 Enforce coding standards & best practices
- 🎯 Generate consistent code structure
- 🤖 AI-friendly (easy to understand & modify)
- ✅ Reduce human error

**Filosofi**: "Convention over Configuration" - CLI tool mengikuti conventions yang sudah established di PROJECT-BRIEF.md dan AI-RULES.md.

// Auto-generated by @platform-cms/cli
// Generated on: 2024-01-08 10:00:00

import { Injectable, NotFoundException, ConflictException } from '@nestjs/common';
import { ProductsRepository } from './products.repository';
import { CreateProductDto } from './dto/create-product.dto';
import { UpdateProductDto } from './dto/update-product.dto';
import { FindAllProductsDto } from './dto/find-all-products.dto';
import { ProductResponseDto } from './dto/product-response.dto';
import { AuditService } from '@/common/services/audit.service';

@Injectable()
export class ProductsService {
  constructor(
    private readonly productsRepository: ProductsRepository,
    private readonly auditService: AuditService,
  ) {}

  async create(dto: CreateProductDto, userId: number): Promise<ProductResponseDto> {
    // Check slug uniqueness
    const existing = await this.productsRepository.findBySlug(dto.slug);
    if (existing) {
      throw new ConflictException({
        code: 'SLUG_ALREADY_EXISTS',
        message: 'Slug sudah digunakan',
      });
    }

    // Create product
    const product = await this.productsRepository.create({
      ...dto,
      createdBy: userId,
      updatedBy: userId,
    });

    // Audit log
    await this.auditService.log({
      userId,
      action: 'create',
      resource: 'products',
      resourceId: product.id,
      newValues: product,
    });

    return new ProductResponseDto(product);
  }

  async findAll(dto: FindAllProductsDto): Promise<{
    data: ProductResponseDto[];
    meta: any;
  }> {
    const { data, total } = await this.productsRepository.findAll(dto);

    return {
      data: data.map(item => new ProductResponseDto(item)),
      meta: {
        pagination: {
          page: dto.page,
          perPage: dto.perPage,
          total,
          totalPages: Math.ceil(total / dto.perPage),
        },
      },
    };
  }

  async findOne(id: number): Promise<ProductResponseDto> {
    const product = await this.productsRepository.findById(id);

    if (!product) {
      throw new NotFoundException({
        code: 'PRODUCT_NOT_FOUND',
        message: 'Product tidak ditemukan',
      });
    }

    return new ProductResponseDto(product);
  }

  async update(
    id: number,
    dto: UpdateProductDto,
    userId: number,
  ): Promise<ProductResponseDto> {
    const existing = await this.findOne(id);

    const updated = await this.productsRepository.update(id, {
      ...dto,
      updatedBy: userId,
    });

    // Audit log
    await this.auditService.log({
      userId,
      action: 'update',
      resource: 'products',
      resourceId: id,
      oldValues: existing,
      newValues: updated,
    });

    return new ProductResponseDto(updated);
  }

  async remove(id: number, userId: number): Promise<void> {
    const existing = await this.findOne(id);

    // Soft delete
    await this.productsRepository.softDelete(id, userId);

    // Audit log
    await this.auditService.log({
      userId,
      action: 'delete',
      resource: 'products',
      resourceId: id,
      oldValues: existing,
    });
  }
}

### 2.3 CRUD Generation

#### Command: `cms generate crud <name>`

**Alias**: `cms g crud <name>`, `cms g c <name>`

**Description**: Generate complete CRUD (Create, Read, Update, Delete) for a resource

**Usage**:
```bash
cms generate crud products

# With fields specification
cms generate crud products --fields="name:string,price:number,description:text"
```

**Options**:
- `--fields` - Comma-separated field definitions (name:type)
- `--relations` - Define relationships (belongsTo, hasMany)
- `--tenant` - Add tenant isolation
- `--soft-delete` - Add soft delete
- `--search` - Add search functionality
- `--export` - Add export to CSV/Excel

**Field Types**:
- `string` → varchar(255)
- `text` → text
- `number` → bigint
- `decimal` → decimal(10,2)
- `boolean` → boolean
- `date` → timestamp
- `email` → varchar(255) with email validation
- `url` → varchar(500) with URL validation
- `json` → jsonb

**Example**:
```bash
cms generate crud articles --fields="title:string,slug:string,content:text,published:boolean,author_id:number" --relations="belongsTo:users" --search --tenant
```

**What it generates**:
- Module (controller, service, repository, entity)
- DTOs with Zod validation
- API endpoints (GET, POST, PUT, DELETE)
- Tests (unit + integration)
- Migration file
- API documentation (Swagger decorators)

### 4.2 Generator Base Class

```typescript
// src/generators/base.generator.ts
import * as fs from 'fs-extra';
import * as path from 'path';
import * as Handlebars from 'handlebars';
import * as prettier from 'prettier';
import chalk from 'chalk';
import ora from 'ora';

export abstract class BaseGenerator {
  protected templateDir: string;
  protected outputDir: string;
  
  constructor(templateDir: string, outputDir: string) {
    this.templateDir = templateDir;
    this.outputDir = outputDir;
  }
  
  /**
   * Load and compile Handlebars template
   */
  protected loadTemplate(templateName: string): HandlebarsTemplateDelegate {
    const templatePath = path.join(this.templateDir, templateName);
    const templateContent = fs.readFileSync(templatePath, 'utf-8');
    return Handlebars.compile(templateContent);
  }
  
  /**
   * Generate file from template
   */
  protected async generateFile(
    templateName: string,
    outputPath: string,
    data: any,
  ): Promise<void> {
    const spinner = ora(`Generating ${outputPath}`).start();
    
    try {
      // Compile template
      const template = this.loadTemplate(templateName);
      let content = template(data);
      
      // Format with Prettier
      const ext = path.extname(outputPath);
      if (['.ts', '.tsx', '.js', '.jsx'].includes(ext)) {
        content = await prettier.format(content, {
          parser: 'typescript',
          singleQuote: true,
          trailingComma: 'all',
          tabWidth: 2,
        });
      }
      
      // Ensure directory exists
      const dir = path.dirname(outputPath);
      await fs.ensureDir(dir);
      
      // Write file
      await fs.writeFile(outputPath, content);
      
      spinner.succeed(chalk.green(`✓ Generated ${outputPath}`));
    } catch (error) {
      spinner.fail(chalk.red(`✗ Failed to generate ${outputPath}`));
      throw error;
    }
  }
  
  /**
   * Transform string to different cases
   */
  protected transformCase(str: string) {
    return {
      kebab: this.toKebabCase(str),
      camel: this.toCamelCase(str),
      pascal: this.toPascalCase(str),
      snake: this.toSnakeCase(str),
      upper: str.toUpperCase(),
      lower: str.toLowerCase(),
    };
  }
  
  private toKebabCase(str: string): string {
    return str
      .replace(/([a-z])([A-Z])/g, '$1-$2')
      .replace(/[\s_]+/g, '-')
      .toLowerCase();
  }
  
  private toCamelCase(str: string): string {
    return str
      .replace(/[-_\s]+(.)?/g, (_, c) => (c ? c.toUpperCase() : ''))
      .replace(/^[A-Z]/, (c) => c.toLowerCase());
  }
  
  private toPascalCase(str: string): string {
    const camel = this.toCamelCase(str);
    return camel.charAt(0).toUpperCase() + camel.slice(1);
  }
  
  private toSnakeCase(str: string): string {
    return str
      .replace(/([a-z])([A-Z])/g, '$1_$2')
      .replace(/[\s-]+/g, '_')
      .toLowerCase();
  }
  
  /**
   * Parse field definitions
   */
  protected parseFields(fieldsStr: string): Array<{
    name: string;
    type: string;
    required: boolean;
  }> {
    if (!fieldsStr) return [];
    
    return fieldsStr.split(',').map((field) => {
      const [name, type = 'string'] = field.trim().split(':');
      return {
        name: name.trim(),
        type: type.trim(),
        required: true, // Default required
      };
    });
  }
  
  /**
   * Map field type to database type
   */
  protected mapFieldType(type: string): {
    dbType: string;
    tsType: string;
    zodType: string;
  } {
    const typeMap: Record<string, any> = {
      string: {
        dbType: "varchar('name', { length: 255 })",
        tsType: 'string',
        zodType: 'z.string().max(255)',
      },
      text: {
        dbType: 'text()',
        tsType: 'string',
        zodType: 'z.string()',
      },
      number: {
        dbType: "bigint('name', { mode: 'number' })",
        tsType: 'number',
        zodType: 'z.number().int()',
      },
      decimal: {
        dbType: "decimal('name', { precision: 10, scale: 2 })",
        tsType: 'number',
        zodType: 'z.number()',
      },
      boolean: {
        dbType: 'boolean()',
        tsType: 'boolean',
        zodType: 'z.boolean()',
      },
      date: {
        dbType: "timestamp('name', { withTimezone: true })",
        tsType: 'Date',
        zodType: 'z.date()',
      },
      email: {
        dbType: "varchar('name', { length: 255 })",
        tsType: 'string',
        zodType: 'z.string().email()',
      },
      url: {
        dbType: "varchar('name', { length: 500 })",
        tsType: 'string',
        zodType: 'z.string().url()',
      },
      json: {
        dbType: 'jsonb()',
        tsType: 'any',
        zodType: 'z.any()',
      },
    };
    
    return typeMap[type] || typeMap.string;
  }
}
```
